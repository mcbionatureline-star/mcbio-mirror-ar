<script type="module">

import * as THREE from 'https://cdn.jsdelivr.net/npm/three@0.136.0/build/three.module.js';

 

const start = async () => {

const mindarThree = new window.MINDAR.FACE.MindARThree({

container: document.body,

});

 

const {renderer, scene, camera} = mindarThree;

 

const light = new THREE.HemisphereLight(0xffffff, 0xbbbbff, 1);

scene.add(light);

 

// Anchor viso (zona bocca = 13)

const anchor = mindarThree.addAnchor(13);

 

// Texture cosmetica (puoi sostituire con tua immagine)

const texture = new THREE.TextureLoader().load(

'https://i.imgur.com/8zQZ5lK.png'

);

 

const material = new THREE.MeshBasicMaterial({

map: texture,

transparent: true

});

 

const geometry = new THREE.PlaneGeometry(0.4, 0.25);

const mesh = new THREE.Mesh(geometry, material);

 

anchor.group.add(mesh);

 

await mindarThree.start();

 

renderer.setAnimationLoop(() => {

renderer.render(scene, camera);

});

};

 

start();

 

// Cambia effetto (demo)

window.changeEffect = () => {

alert("Qui puoi cambiare texture prodotto (rossetto, crema, blush)");

};

 

</script>
